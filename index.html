// v9 const VERSION = 'sp360-v9'; const BASE = self.registration.scope; // e.g. https://.github.io/Service-pro-360/ const ASSETS = [ `${BASE}`, `${BASE}index.html`, `${BASE}app.js?v=9`, `${BASE}manifest.json`, ]; // Install: warm the cache self.addEventListener('install', (e) => { e.waitUntil( caches.open(VERSION) .then(c => c.addAll(ASSETS)) .then(() => self.skipWaiting()) ); }); // Activate: clear old caches self.addEventListener('activate', (e) => { e.waitUntil( caches.keys() .then(keys => Promise.all(keys.map(k => (k === VERSION ? null : caches.delete(k))))) .then(() => self.clients.claim()) ); }); // Respond: network-first, then cache, fall back to error self.addEventListener('fetch', (e) => { e.respondWith((async () => { try { const net = await fetch(e.request); (await caches.open(VERSION)).put(e.request, net.clone()); return net; } catch { return (await caches.match(e.request)) || Response.error(); } })()); }); // Let the page know when weâ€™re ready + expose version self.addEventListener('message', (e) => { if (e.data && e.data.type === 'PING') { e.source.postMessage({type: 'SW_READY', version: VERSION}); } });
